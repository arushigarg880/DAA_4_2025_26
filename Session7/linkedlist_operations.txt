#include <iostream>
using namespace std;
class node{
public:
    int data;
    node* next;
    node(int value){
        data = value;
        next = NULL;
}};
void insert(node*&head,int value){
    if(head==NULL){
        head = new node(value);
        return;
        
    }
    node*temp = head;
    
    while (temp->next!=NULL){
        temp = temp->next;
    }
    temp->next = new node(value);
}
void display(node*head){
    node*temp = head;
while (temp!=NULL){
    cout<<temp->data<<" ";
    temp = temp->next;
}

}
void delete1(node*&head,int val){
    if(head->data==val){
        node*temp=head;
        head=head->next;
        delete temp;
        
    }
    node*temp=head;
    
    while(temp->next->data!=val&&temp->next!=NULL){
        temp=temp->next;
    }
    if(temp->next==NULL)return;
    node*ptr=temp->next;
    temp->next=ptr->next;
    delete ptr;
    
}
int main() {
  
node*head = new node(12);
insert(head,12);
insert(head,30);
insert(head,79);
insert(head,50);
insert(head,51);
display(head);
delete1(head,30);
delete1(head,51);
cout<<"\n";
display(head);

    return 0;
}
